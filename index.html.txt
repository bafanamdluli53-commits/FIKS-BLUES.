    <script>
        const app = {
            charts: {},
            wallet: JSON.parse(localStorage.getItem('p_wallet')) || { balance: 10000, position: null },
            signalHistory: JSON.parse(localStorage.getItem('p_signals')) || [],

            async init() {
                this.initCharts();
                this.signalHistory.forEach(s => this.renderSignalUI(s));
                this.updateWalletUI();
                await this.analyze();
                setInterval(() => this.analyze(), 10000); // Auto-refresh every 10s
            },

            initCharts() {
                const ctx = document.getElementById('priceChart').getContext('2d');
                this.charts.price = new Chart(ctx, {
                    type: 'line',
                    data: { labels: [], datasets: [
                        { label: 'Price', data: [], borderColor: '#3b82f6', tension: 0.2, pointRadius: 0, fill: true, backgroundColor: 'rgba(59, 130, 246, 0.1)' },
                        { label: 'EMA 9', data: [], borderColor: '#8b5cf6', borderWidth: 1, pointRadius: 0 }
                    ]},
                    options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { x: { display: false }, y: { grid: { color: '#374151' } } } }
                });
            },

            async analyze() {
                const symbol = document.getElementById('symbolInput').value.toUpperCase();
                try {
                    const res = await fetch(`https://api.binance.com/api/v3/klines?symbol=${symbol}&interval=1h&limit=50`);
                    const data = await res.json();
                    const prices = data.map(d => parseFloat(d[4]));
                    const lastPrice = prices[prices.length - 1];

                    const ema9 = this.calcEMA(prices, 9);
                    this.updateChart(prices, ema9);
                    this.checkSignals(symbol, prices, ema9);
                    
                    document.getElementById('currentPrice').innerText = `$${lastPrice.toLocaleString()}`;
                    this.updateWalletUI();
                } catch (e) { console.error(e); }
            },

            calcEMA(data, p) {
                const k = 2 / (p + 1);
                let ema = [data[0]];
                for (let i = 1; i < data.length; i++) ema.push(data[i] * k + ema[i-1] * (1 - k));
                return ema;
            },

            updateChart(prices, ema) {
                this.charts.price.data.labels = prices.map((_, i) => i);
                this.charts.price.data.datasets[0].data = prices;
                this.charts.price.data.datasets[1].data = ema;
                this.charts.price.update('none');
            },

            checkSignals(symbol, prices, ema) {
                const cur = prices[prices.length - 1];
                const prev = prices[prices.length - 2];
                const cEma = ema[ema.length - 1];

                if (prev < cEma && cur > cEma) this.logSignal(symbol, 'buy', cur, 'EMA Cross Up');
                if (prev > cEma && cur < cEma) this.logSignal(symbol, 'sell', cur, 'EMA Cross Down');
            },

            logSignal(symbol, type, price, reason) {
                if(this.signalHistory[0]?.price === price) return;
                const sig = { symbol, type, price, reason, time: new Date().toLocaleTimeString() };
                this.signalHistory.unshift(sig);
                if(this.signalHistory.length > 20) this.signalHistory.pop();
                localStorage.setItem('p_signals', JSON.stringify(this.signalHistory));
                this.renderSignalUI(sig);
            },

            renderSignalUI(s) {
                const list = document.getElementById('signalList');
                const html = `<div class="signal-item ${s.type}"><strong>${s.type.toUpperCase()}</strong> ${s.symbol}<br><small>$${s.price} | ${s.time}</small></div>`;
                list.insertAdjacentHTML('afterbegin', html);
            },

            executeTrade(type) {
                const price = parseFloat(document.getElementById('currentPrice').innerText.replace(/[^0-9.]/g, ''));
                if (type === 'buy' && !this.wallet.position) {
                    this.wallet.position = { entry: price, qty: 2000 / price };
                    this.wallet.balance -= 2000;
                } else if (type === 'sell' && this.wallet.position) {
                    this.wallet.balance += (this.wallet.position.qty * price);
                    this.wallet.position = null;
                }
                localStorage.setItem('p_wallet', JSON.stringify(this.wallet));
                this.updateWalletUI();
            },

            updateWalletUI() {
                const price = parseFloat(document.getElementById('currentPrice').innerText.replace(/[^0-9.]/g, ''));
                document.getElementById('walletDisplay').innerText = `$${this.wallet.balance.toFixed(2)}`;
                if (this.wallet.position) {
                    const pnl = (price - this.wallet.position.entry) * this.wallet.position.qty;
                    document.getElementById('pnlDisplay').innerText = `${pnl >= 0 ? '+' : ''}$${pnl.toFixed(2)}`;
                    document.getElementById('pnlDisplay').style.color = pnl >= 0 ? 'var(--accent-green)' : 'var(--accent-red)';
                } else {
                    document.getElementById('pnlDisplay').innerText = "$0.00";
                }
            },

            exportToCSV() {
                let csv = "Time,Symbol,Type,Price,Reason\n" + this.signalHistory.map(s => `${s.time},${s.symbol},${s.type},${s.price},${s.reason}`).join("\n");
                const blob = new Blob([csv], { type: 'text/csv' });
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url; a.download = 'trades.csv'; a.click();
            }
        };

        window.onload = () => app.init();
    </script>
